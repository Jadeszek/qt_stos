\section{Czym jest Qt?}
\subsection{Krótko o Qt}
\frame{\frametitle{Krótko o Qt}

\begin{itemize}
\item Framework do tworzenia aplikacji w C++ (i nie tylko - PyQt, QtJambi)
\item Kompilacja kodu na różne platformy bez konieczności zmian
\item Tworzenie natywnych aplikacji == wydajność
\item Niezależne od systemu GUI
\end{itemize} 
}

\subsection{Moduły}
\frame{\frametitle{Moduły}
\begin{itemize}
\item Qt Core - podstawowy, wspołbieżność, wątki, wejście-wyjście  
\item Qt GUI, Qt Widgets - budowa GUI, kontrolki
\item Qt Multimedia - dźwięk, obraz, wideo, łączność radiowa, kamera
\item Qt Network - programowanie sieciowe
\item Qt QML, Qt Quick - obsługa języka QML i JavaScript oraz GUI doń
\item Qt SQL - obsługa baz danych
\item Qt Test - framework do testowania aplikacji
\item Qt WebKit - silnik do renderowania stron WWW
\end{itemize} 
}
